```prisma
model EmployeeDay {
  id            String      @id @default(cuid())
  employeeId    String
  dayId         String
  batchStrainId String
  amount        Float
  hours         Float?
  employee      User        @relation(fields: [employeeId], references: [id])
  day           Day         @relation(fields: [dayId], references: [id], onDelete: Cascade)
  batchStrain   BatchStrain @relation(fields: [batchStrainId], references: [id])
  @@map("employees_days")
}

model Day {
  id           String        @id @default(cuid())
  batchId      String
  batchDay     Int
  isSubmitted  Boolean       @default(false)
  batch        Batch         @relation(fields: [batchId], references: [id], onDelete: Cascade)
  employeeDays EmployeeDay[]
  @@unique([batchId, batchDay])
}

model Batch {
  id           String        @id @default(cuid())
  number       Int
  status       BatchStatus   @default(INACTIVE)
  startDate    DateTime?
  endDate      DateTime?
  batchStrains BatchStrain[]
  days         Day[]
}

model BatchStrain {
  id           String        @id @default(cuid())
  batchId      String
  strainId     String
  strain       Strain        @relation(fields: [strainId], references: [id])
  employeeDays EmployeeDay[]
  @@unique([batchId, strainId])
}

model Strain {
  id           String        @id @default(cuid())
  name         String
}
```

    2. **`fetchEmployeePerformanceData(userId: string): Promise<EmployeeDayWithRelations[]>`**
    Prisma query — `prisma.employeeDay.findMany` with:
    - `where: { employeeId: userId, day: { isSubmitted: true } }`
    - `include: { day: { include: { batch: true } }, batchStrain: { include: { strain: true } } }`
    - `orderBy: { day: { batchDay: 'desc' } }`

    3. **`getLatestSubmittedDay(entries: EmployeeDayWithRelations[]): DashboardCardData | null`**
    Pure function. If entries empty, return null. Otherwise:
    - Find the most recent batch (highest `batch.number`), then within it the highest `batchDay`
    - Filter entries for that day (matching `dayId`)
    - Sum `amount` across those entries
    - Compute calendar date: `new Date(batch.startDate.getTime() + (day.batchDay - 1) * 86_400_000)`
    - Format date with `toLocaleDateString('en-US', { month: 'long', day: 'numeric' })`
    - Return `{ totalGrams, dateLabel, dayNumber: day.batchDay }`

    4. **`groupByBatch(entries: EmployeeDayWithRelations[]): BatchSummary[]`**
    Pure function. Group by `day.batch.id`, sum amounts per batch, return sorted by `batchNumber` descending. Convert dates to ISO strings.

    5. **`groupByStrain(entries: EmployeeDayWithRelations[]): Map<string, number>`**
    Pure function. Aggregate `amount` per `batchStrain.strain.name` across all entries.

    6. **`groupDaysByBatch(entries: EmployeeDayWithRelations[], batchId: string): DayDetail[]`**
    Pure function. Filter entries for given batchId, group by dayId, compute totalGrams per day, collect strainEntries per day, sort by dayNumber descending. Compute dateLabel from batch.startDate + (batchDay - 1) days.

    7. **`formatCalendarDate(startDate: Date | null, batchDay: number): string`**
    Pure helper. Returns formatted string like "March 24" or "Day N" if startDate is null.
